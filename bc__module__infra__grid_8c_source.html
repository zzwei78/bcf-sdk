120 bool bc_module_infra_grid_get_temperatures_celsius(bc_module_infra_grid_t *self, float *values)
121 {

123  {
124  return false;
125  }
126 
127  for (int i = 0; i < 64 ;i++)
128  {
129  float temperature;

136  else

139  }
140 
141  values[i] = temperature;
142  }

154 static void _bc_module_infra_grid_task_measure(void *param)
155 {
156  bc_module_infra_grid_t *self = param;
157 
158  start:
159 
160  switch (self->_state)
161  {
162  case BC_MODULE_INFRA_GRID_STATE_ERROR:
163  {
164  self->_temperature_valid = false;
165 
166  self->_measurement_active = false;
167 
168  if (self->_event_handler != NULL)
169  {
170  self->_event_handler(self, BC_MODULE_INFRA_GRID_EVENT_ERROR, self->_event_param);
171  }
172 
173  self->_state = BC_MODULE_INFRA_GRID_STATE_INITIALIZE;
174 
175  return;
176  }
177  case BC_MODULE_INFRA_GRID_STATE_INITIALIZE:
178  {
179  self->_state = BC_MODULE_INFRA_GRID_STATE_ERROR;
180 
181  if (bc_tca9534a_init(&self->_tca9534, BC_I2C_I2C0 , 0x23))
182  {
183  self->_revision = BC_MODULE_INFRA_GRID_REVISION_R1_1;
184  }
185  else
186  {
187  self->_revision = BC_MODULE_INFRA_GRID_REVISION_R1_0;
188  }
189 
190  if (self->_revision == BC_MODULE_INFRA_GRID_REVISION_R1_1)
191  {
192  if (!bc_tca9534a_set_pin_direction(&self->_tca9534, _BC_MODULE_INFRA_GRID_PIN_POWER , BC_TCA9534A_PIN_DIRECTION_OUTPUT))
193  {
194  goto start;
195  }
196  if (!bc_tca9534a_write_pin(&self->_tca9534, _BC_MODULE_INFRA_GRID_PIN_POWER , 1))
197  {
198  goto start;
199  }
200  }
201 
202  // Update sleep flag
203  if (self->_enable_sleep != self->_cmd_sleep)
204  {
205  self->_enable_sleep = self->_cmd_sleep;
206  }
207 
208  if (self->_enable_sleep)
209  {
210  if (self->_revision == BC_MODULE_INFRA_GRID_REVISION_R1_0)
211  {
212  // Sleep mode
213  if (!bc_i2c_memory_write_8b(self->_i2c_channel, self->_i2c_address, _BC_AMG88xx_PCLT, 0x10))
214  {
215  goto start;
216  }
217  }
218  else if (self->_revision == BC_MODULE_INFRA_GRID_REVISION_R1_1)
219  {
220  // Revision 1.1 - Disconnect power to Infra Grid Module
221  if (!bc_tca9534a_write_pin(&self->_tca9534, _BC_MODULE_INFRA_GRID_PIN_POWER , 0))
222  {
223  goto start;
224  }
225  }
226  }
227  else
228  {
229  // Set 10 FPS
230  if (!bc_i2c_memory_write_8b (self->_i2c_channel, self->_i2c_address, _BC_AMG88xx_FPSC, 0x00))
231  {
232  goto start;
233  }
234 
235  // Diff interrpt mode, INT output reactive
236  bc_i2c_memory_write_8b(self->_i2c_channel, self->_i2c_address, _BC_AMG88xx_INTC, 0x00);
237  // Moving average output mode active
238  bc_i2c_memory_write_8b(self->_i2c_channel, self->_i2c_address, _BC_AMG88xx_AVG, 0x50);
239  bc_i2c_memory_write_8b(self->_i2c_channel, self->_i2c_address, _BC_AMG88xx_AVG, 0x45);
240  bc_i2c_memory_write_8b(self->_i2c_channel, self->_i2c_address, _BC_AMG88xx_AVG, 0x57);
241  bc_i2c_memory_write_8b(self->_i2c_channel, self->_i2c_address, _BC_AMG88xx_AVE, 0x20);
242  bc_i2c_memory_write_8b(self->_i2c_channel, self->_i2c_address, _BC_AMG88xx_AVG, 0x00);
243  }
244 
245  self->_state = BC_MODULE_INFRA_GRID_STATE_MODE_CHANGE; //BC_MODULE_INFRA_GRID_STATE_MEASURE;
246 
247  self->_tick_ready = bc_tick_get() + _BC_MODULE_INFRA_GRID_DELAY_INITIALIZATION;
248 
249  if (self->_measurement_active)
250  {
251  bc_scheduler_plan_current_absolute(self->_tick_ready);
252  }
253 
254  return;
255  }
256  case BC_MODULE_INFRA_GRID_STATE_MODE_CHANGE:
257  {
258  // Skip wakeup commands in case of fast reading
259  if (!self->_enable_sleep)
260  {
261  self->_state = BC_MODULE_INFRA_GRID_STATE_READ;
262  goto start;
263  }
264 
265  if (self->_revision == BC_MODULE_INFRA_GRID_REVISION_R1_0)
266  {
267  // Revision 1.0 - The module is already powered up
268  self->_state = BC_MODULE_INFRA_GRID_STATE_POWER_UP;
269  goto start;
270  }
271  else if (self->_revision == BC_MODULE_INFRA_GRID_REVISION_R1_1)
272  {
273  // Revision 1.1 - Enable power for Infra Grid Module
274  if (!bc_tca9534a_write_pin(&self->_tca9534, _BC_MODULE_INFRA_GRID_PIN_POWER , 1))
275  {
276  self->_state = BC_MODULE_INFRA_GRID_STATE_ERROR;
277  goto start;
278  }
279  self->_state = BC_MODULE_INFRA_GRID_STATE_POWER_UP;
280  bc_scheduler_plan_current_from_now(_BC_MODULE_INFRA_GRID_DELAY_POWER_UP);
281  }
282 
283  return;
284  }
285  case BC_MODULE_INFRA_GRID_STATE_POWER_UP:
286  {
287  self->_state = BC_MODULE_INFRA_GRID_STATE_ERROR;
288 
289  // Normal Mode
290  if (!bc_i2c_memory_write_8b(self->_i2c_channel, self->_i2c_address, _BC_AMG88xx_PCLT, 0x00))
291  {
292  goto start;
293  }
294  self->_state = BC_MODULE_INFRA_GRID_STATE_INITIAL_RESET;
295  bc_scheduler_plan_current_from_now(_BC_MODULE_INFRA_GRID_DELAY_MODE_CHANGE);
296  return;
297  }
298  case BC_MODULE_INFRA_GRID_STATE_INITIAL_RESET:
299  {
300  self->_state = BC_MODULE_INFRA_GRID_STATE_ERROR;
301 
302  // Write initial reset
303  if (!bc_i2c_memory_write_8b(self->_i2c_channel, self->_i2c_address, _BC_AMG88xx_RST, 0x3f))
304  {
305  goto start;
306  }
307 
308  self->_state = BC_MODULE_INFRA_GRID_STATE_FLAG_RESET;
309  bc_scheduler_plan_current_from_now(_BC_MODULE_INFRA_GRID_DELAY_INITIAL_RESET);
310  return;
311  }
312  case BC_MODULE_INFRA_GRID_STATE_FLAG_RESET:
313  {
314  self->_state = BC_MODULE_INFRA_GRID_STATE_ERROR;
315 
316  // Write flag reset
317  if (!bc_i2c_memory_write_8b(self->_i2c_channel, self->_i2c_address, _BC_AMG88xx_RST, 0x30))
318  {
319  goto start;
320  }
321 
322  self->_state = BC_MODULE_INFRA_GRID_STATE_MEASURE;
323  bc_scheduler_plan_current_from_now(_BC_MODULE_INFRA_GRID_DELAY_FLAG_RESET);
324  return;
325  }
326  case BC_MODULE_INFRA_GRID_STATE_MEASURE:
327  {
328  self->_state = BC_MODULE_INFRA_GRID_STATE_ERROR;
329 
330  if (!bc_i2c_memory_write_8b(self->_i2c_channel, self->_i2c_address, _BC_AMG88xx_FPSC, 0x01))
331  {
332  goto start;
333  }
334 
335  // Diff interrpt mode, INT output reactive
336  bc_i2c_memory_write_8b(self->_i2c_channel, self->_i2c_address, _BC_AMG88xx_INTC, 0x00);
337  // Moving average output mode active
338  bc_i2c_memory_write_8b(self->_i2c_channel, self->_i2c_address, _BC_AMG88xx_AVG, 0x50);
339  bc_i2c_memory_write_8b(self->_i2c_channel, self->_i2c_address, _BC_AMG88xx_AVG, 0x45);
340  bc_i2c_memory_write_8b(self->_i2c_channel, self->_i2c_address, _BC_AMG88xx_AVG, 0x57);
341  bc_i2c_memory_write_8b(self->_i2c_channel, self->_i2c_address, _BC_AMG88xx_AVE, 0x20);
342  bc_i2c_memory_write_8b(self->_i2c_channel, self->_i2c_address, _BC_AMG88xx_AVG, 0x00);
343 
344  self->_state = BC_MODULE_INFRA_GRID_STATE_READ;
345 
346  bc_scheduler_plan_current_from_now(_BC_MODULE_INFRA_GRID_DELAY_MEASUREMENT);
347 
348  return;
349  }
350  case BC_MODULE_INFRA_GRID_STATE_READ:
351  {
352  self->_state = BC_MODULE_INFRA_GRID_STATE_ERROR;
353 
354  bc_i2c_memory_transfer_t transfer;
355  transfer.device_address = self->_i2c_address;
356  transfer.memory_address = _BC_AMG88xx_T01L;
357  transfer.buffer = self->_sensor_data;
358  transfer.length = 64 * 2;
359 
360  if (!bc_i2c_memory_read(self->_i2c_channel, &transfer))
361  {
362  goto start;
363  }
364 
365  // Enable sleep mode
366  if (self->_enable_sleep)
367  {
368  // Sleep Mode
369  if (self->_revision == BC_MODULE_INFRA_GRID_REVISION_R1_0)
370  {
371  // Sleep mode
372  if (!bc_i2c_memory_write_8b(self->_i2c_channel, self->_i2c_address, _BC_AMG88xx_PCLT, 0x10))
373  {
374  goto start;
375  }
376  }
377  else if (self->_revision == BC_MODULE_INFRA_GRID_REVISION_R1_1)
378  {
379  // Revision 1.1 has power disconnect
380  bc_tca9534a_write_pin(&self->_tca9534, _BC_MODULE_INFRA_GRID_PIN_POWER , 0);
381  }
382  }
383 
384  self->_temperature_valid = true;
385  self->_state = BC_MODULE_INFRA_GRID_STATE_UPDATE;
386 
387  goto start;
388  }
389  case BC_MODULE_INFRA_GRID_STATE_UPDATE:
390  {
391  self->_measurement_active = false;
392 
393  if (self->_event_handler != NULL)
394  {
395  self->_event_handler(self, BC_MODULE_INFRA_GRID_EVENT_UPDATE, self->_event_param);
396  }
397 
398  // Update sleep flag
399  if (self->_enable_sleep != self->_cmd_sleep)
400  {
401  self->_enable_sleep = self->_cmd_sleep;
402  }
403 
404  self->_state = BC_MODULE_INFRA_GRID_STATE_MODE_CHANGE;
405  return;
406  }
407  default:
408  {
409  self->_state = BC_MODULE_INFRA_GRID_STATE_ERROR;
410  goto start;
411  }
412  }
413 }
