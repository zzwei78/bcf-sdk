155  self->_event_handler(self, BC_TMP112_EVENT_ERROR, self->_event_param);
156  }
157 
158  self->_state = BC_TMP112_STATE_INITIALIZE;
159

161  }
162  case BC_TMP112_STATE_INITIALIZE:
163  {

165 
166  if (!bc_i2c_memory_write_16b(self->_i2c_channel, self->_i2c_address, 0x01, 0x0180))
167  {
168  goto start;
169  }
170 
171  self->_state = BC_TMP112_STATE_MEASURE;
172

174 
175  if (self->_measurement_active)

177  bc_scheduler_plan_current_absolute(self->_tick_ready);
178  }
179

181  }
182  case BC_TMP112_STATE_MEASURE:
183  {
